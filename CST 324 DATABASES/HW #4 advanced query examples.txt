	SELECT 	CUS_CODE, INVOICE.INV_NUMBER, INV_DATE
		P_DESCRIPT, LINE_UNITS, LINE_PRICE
	FROM 	INVOICE, LINE, PRODUCT
	WHERE	(INVOICE.INV_NUMBER = LINE.INV_NUMBER)	AND
		(LINE.P_CODE = PRODUCT.P_CODE)
	ORDER BY CUS_CODE;

PROBLEMS #19 (PAGE 309)

	COMMAND 1:
	CREATE PROB19 VIEW AS
	SELECT 	CUS_CODE, INV_NUMBER, P_DESCRIPT,
		LINE_UNITS AS [UNITS BOUGHT],
		LINE_PRICE AS [UNIT PRICE],
		LINE_UNITS * LINE_PRICE AS [SUBTOTAL]
	FROM 	INVOICE, LINE, PRODUCT
	WHERE	(INVOICE.INV_NUMBER = LINE.INV_NUMBER)	AND
		(LINE.P_CODE = PRODUCT.P_CODE)
	ORDER BY CUS_CODE;

	COMMAND2:
	CREATE VIEW AS TOTALPURCH AS
	SELECT CUS_CODE, SUM(SUBTOTAL) AS [TOTAL PURCHASES]
	FROM	PROB19
	GROUP BY CUS_CODE;

	COMMAND 3:
	SELECT 	TOTALPURCH.CUS_CODE, [TOTAL PURCHASES], CUS_BALANCE
	FROM	TOTALPURCH, CUSTOMER
	WHERE	TOTALPURCH.CUST_CODE = CUSTOMER.CUS_CODE;
